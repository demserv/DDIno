```powershell
param(
    [string]$Root = ".",
    [string]$Out = "auditoria_codigo_proprio.txt"
)

$includeExtensions = @(".c", ".h", ".cpp", ".hpp", ".md", ".txt", ".cmake")

$excludeDirs = @(
    "managed_components",
    "build",
    ".git",
    ".vscode",
    "node_modules"
)

$excludeExtensions = @(
    ".png", ".jpg", ".jpeg", ".gif", ".bmp",
    ".ttf", ".bin", ".elf", ".map", ".o", ".a", ".zip"
)

if (Test-Path $Out) {
    Remove-Item $Out -Force
}

$files = Get-ChildItem -Path $Root -Recurse -File | Where-Object {
    $full = $_.FullName
    $ext = $_.Extension.ToLowerInvariant()

    foreach ($dir in $excludeDirs) {
        if ($full -match "[\\/]$([regex]::Escape($dir))[\\/]") {
            return $false
        }
    }

    if ($excludeExtensions -contains $ext) {
        return $false
    }

    return $includeExtensions -contains $ext
} | Sort-Object FullName

"TOTAL_ARQUIVOS_AUDITAVEIS=$($files.Count)" |
    Out-File -FilePath $Out -Encoding UTF8

foreach ($f in $files) {
    "`n===== FILE: $($f.FullName) =====`n" |
        Out-File -FilePath $Out -Append -Encoding UTF8

    Get-Content $f.FullName -Raw -ErrorAction SilentlyContinue |
        Out-File -FilePath $Out -Append -Encoding UTF8
}

Write-Host "Arquivo gerado: $Out"
Write-Host "Total de arquivos auditáveis: $($files.Count)"
```

```python
from pathlib import Path
import sys

forbidden = ["PZEM-004T v3.0"]

roots = [
    Path("firmware/main"),
    Path("firmware/docs"),
    Path("firmware/ARCHITECTURE.md"),
]

failed = False

for root in roots:
    if not root.exists():
        continue

    files = [root] if root.is_file() else list(root.rglob("*"))

    for f in files:
        if not f.is_file():
            continue

        if f.suffix.lower() not in [".c", ".h", ".md", ".txt"]:
            continue

        text = f.read_text(errors="ignore")

        for token in forbidden:
            if token in text:
                print(f"FORBIDDEN TOKEN {token} in {f}")
                failed = True

sys.exit(1 if failed else 0)
```

```c
#include "ui_screen_ato.h"
#include "../ui_theme.h"
#include "../ui_events.h"
#include <stdio.h>

static lv_obj_t *lbl_state;
static lv_obj_t *lbl_sensor;
static lv_obj_t *lbl_pump;
static lv_obj_t *lbl_timer;
static lv_obj_t *lbl_alarm;
static lv_obj_t *btn_refill;
static lv_obj_t *btn_stop;

static const char *ato_state_to_text(ui_ato_state_t s)
{
    switch (s) {
    case UI_ATO_STATE_NORMAL: return "NORMAL";
    case UI_ATO_STATE_REFILLING: return "REFILLING";
    case UI_ATO_STATE_OVERFLOW: return "OVERFLOW";
    case UI_ATO_STATE_BLOCKED: return "BLOCKED";
    case UI_ATO_STATE_ERROR: return "ERROR";
    case UI_ATO_STATE_DISABLED: return "DISABLED";
    default: return "UNKNOWN";
    }
}

static void refill_cb(lv_event_t *e)
{
    (void)e;
    ui_events_emit(UI_EVENT_ATO_START_REFILL);
}

static void stop_cb(lv_event_t *e)
{
    (void)e;
    ui_events_emit(UI_EVENT_ATO_STOP_REFILL);
}

void ui_screen_ato_create(lv_obj_t *parent, ui_root_vm_t *vm)
{
    lv_obj_t *title = lv_label_create(parent);
    lv_label_set_text(title, "ATO / Nivel de Agua");
    lv_obj_set_style_text_font(title, UI_FONT_MEDIUM, 0);
    lv_obj_set_style_text_color(title, UI_COLOR_TEXT_MAIN, 0);
    lv_obj_align(title, LV_ALIGN_TOP_MID, 0, 6);

    lbl_state = lv_label_create(parent);
    lv_obj_set_pos(lbl_state, 20, 45);

    lbl_sensor = lv_label_create(parent);
    lv_obj_set_pos(lbl_sensor, 20, 75);

    lbl_pump = lv_label_create(parent);
    lv_obj_set_pos(lbl_pump, 20, 105);

    lbl_timer = lv_label_create(parent);
    lv_obj_set_pos(lbl_timer, 20, 135);

    lbl_alarm = lv_label_create(parent);
    lv_obj_set_pos(lbl_alarm, 20, 165);

    btn_refill = lv_btn_create(parent);
    lv_obj_set_size(btn_refill, 130, 36);
    lv_obj_set_pos(btn_refill, 40, 220);
    lv_obj_add_event_cb(btn_refill, refill_cb, LV_EVENT_CLICKED, NULL);

    lv_obj_t *l1 = lv_label_create(btn_refill);
    lv_label_set_text(l1, "Iniciar refill");
    lv_obj_center(l1);

    btn_stop = lv_btn_create(parent);
    lv_obj_set_size(btn_stop, 130, 36);
    lv_obj_set_pos(btn_stop, 300, 220);
    lv_obj_add_event_cb(btn_stop, stop_cb, LV_EVENT_CLICKED, NULL);

    lv_obj_t *l2 = lv_label_create(btn_stop);
    lv_label_set_text(l2, "Parar ATO");
    lv_obj_center(l2);

    ui_screen_ato_update(vm);
}

void ui_screen_ato_update(ui_root_vm_t *vm)
{
    if (!vm) return;

    char buf[96];

    snprintf(buf, sizeof(buf), "Estado: %s", ato_state_to_text(vm->ato.state));
    lv_label_set_text(lbl_state, buf);

    snprintf(buf, sizeof(buf), "Sensor agua: %s",
             vm->ato.sensor_water_present ? "OK" : "BAIXO/SECO");
    lv_label_set_text(lbl_sensor, buf);

    snprintf(buf, sizeof(buf), "Bomba ATO: %s",
             vm->ato.pump_commanded_on ? "ON" : "OFF");
    lv_label_set_text(lbl_pump, buf);

    snprintf(buf, sizeof(buf), "Refill: %lu/%lu s",
             (unsigned long)vm->ato.refill_elapsed_s,
             (unsigned long)vm->ato.refill_timeout_s);
    lv_label_set_text(lbl_timer, buf);

    snprintf(buf, sizeof(buf), "Alarme: %s",
             vm->ato.active_alarm[0] ? vm->ato.active_alarm : "nenhum");
    lv_label_set_text(lbl_alarm, buf);

    bool unsafe =
        vm->ato.state == UI_ATO_STATE_OVERFLOW ||
        vm->ato.state == UI_ATO_STATE_BLOCKED ||
        vm->ato.state == UI_ATO_STATE_ERROR ||
        vm->ato.state == UI_ATO_STATE_DISABLED;

    if (unsafe) {
        lv_obj_add_state(btn_refill, LV_STATE_DISABLED);
    } else {
        lv_obj_clear_state(btn_refill, LV_STATE_DISABLED);
    }
```
