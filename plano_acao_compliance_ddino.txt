    cfg->ui.brightness_percent = 80;

    cfg->ui.keypad_enabled = true;
    cfg->ui.carousel_interval_ms = 5000;

    cfg->web.auth_enabled = true;
    cfg->web.session_timeout_s = 900;
    cfg->web.rate_limit_per_minute = 30;

    cfg->crc32 = config_root_crc32(cfg);

esp_err_t config_root_validate(const config_root_t *cfg)
{
    if (!cfg) return ESP_ERR_INVALID_ARG;
    if (cfg->magic != CONFIG_ROOT_MAGIC) return ESP_ERR_INVALID_STATE;
    if (cfg->schema_version != CONFIG_ROOT_SCHEMA_VERSION) return ESP_ERR_INVALID_VERSION;

    if (cfg->thermal.setpoint_c < 15.0f || cfg->thermal.setpoint_c > 35.0f) return ESP_ERR_INVALID_STATE;
    if (cfg->thermal.hysteresis_c <= 0.0f || cfg->thermal.hysteresis_c > 5.0f) return ESP_ERR_INVALID_STATE;
    if (cfg->thermal.critical_low_c >= cfg->thermal.critical_high_c) return ESP_ERR_INVALID_STATE;

    if (cfg->ato.adc_threshold_on > 4095 || cfg->ato.adc_threshold_off > 4095) return ESP_ERR_INVALID_STATE;
    if (cfg->ato.refill_timeout_s == 0) return ESP_ERR_INVALID_STATE;

    if (cfg->ui.brightness_percent > 100) return ESP_ERR_INVALID_STATE;
    if (!cfg->ui.touch_enabled && !cfg->ui.keypad_enabled) return ESP_ERR_INVALID_STATE;

    if (cfg->web.rate_limit_per_minute == 0) return ESP_ERR_INVALID_STATE;

    uint32_t expected = config_root_crc32(cfg);
    if (cfg->crc32 != expected) return ESP_ERR_INVALID_CRC;

    return ESP_OK;
}

uint32_t config_root_crc32(const config_root_t *cfg)
{
    /* Substituir por esp_rom_crc32_le ou função CRC já usada no projeto.
       O campo crc32 deve ser zerado temporariamente durante cálculo. */
    if (!cfg) return 0;
    config_root_t tmp = *cfg;
    tmp.crc32 = 0;

    const uint8_t *p = (const uint8_t *)&tmp;
    uint32_t crc = 0xFFFFFFFFu;
    for (size_t i = 0; i < sizeof(tmp); ++i) {
        crc ^= p[i];
        for (int b = 0; b < 8; ++b) {
            crc = (crc >> 1) ^ (0xEDB88320u & (-(int32_t)(crc & 1u)));
        }
    }
    return ~crc;
}

7. Sugestão de implementação transacional em config_manager.c

```c
esp_err_t config_manager_commit_root_transactional(const config_root_t *cfg)
{
    if (!cfg) return ESP_ERR_INVALID_ARG;

    esp_err_t err = config_root_validate(cfg);
    if (err != ESP_OK) return err;

    /* Estratégia:
       1. Gravar config_root_pending.
       2. Gravar marcador transaction_pending=true.
       3. Copiar pending para active.
       4. Limpar marcador.
       5. Manter backup anterior para rollback. */

    err = nvs_write_blob("cfg", "root_pending", cfg, sizeof(*cfg));
    if (err != ESP_OK) return err;

    bool pending = true;
    err = nvs_write_blob("cfg", "tx_pending", &pending, sizeof(pending));
    if (err != ESP_OK) return err;

    config_root_t old_cfg;
    if (config_manager_load_root(&old_cfg) == ESP_OK) {
        (void)nvs_write_blob("cfg", "root_backup", &old_cfg, sizeof(old_cfg));
    }

    err = nvs_write_blob("cfg", "root_active", cfg, sizeof(*cfg));
    if (err != ESP_OK) return err;

    pending = false;
    err = nvs_write_blob("cfg", "tx_pending", &pending, sizeof(pending));
    if (err != ESP_OK) return err;

    g_config_root_cache = *cfg;
    return ESP_OK;
}
```

Observação: `nvs_write_blob` acima representa helper já existente ou a ser criado com `nvs_open`, `nvs_set_blob`, `nvs_commit`, `nvs_close`.

{
    if (!vm) return;

    char buf[96];

    snprintf(buf, sizeof(buf), "Estado: %s", ato_state_to_text(vm->ato.state));
    lv_label_set_text(lbl_state, buf);

    snprintf(buf, sizeof(buf), "Sensor agua: %s", vm->ato.sensor_water_present ? "OK" : "BAIXO/SECO");
    lv_label_set_text(lbl_sensor, buf);

    snprintf(buf, sizeof(buf), "Bomba ATO: %s", vm->ato.pump_commanded_on ? "ON" : "OFF");
    lv_label_set_text(lbl_pump, buf);

    snprintf(buf, sizeof(buf), "Refill: %lu/%lu s",
             (unsigned long)vm->ato.refill_elapsed_s,
             (unsigned long)vm->ato.refill_timeout_s);

    lv_label_set_text(lbl_alarm, buf);

    bool unsafe = vm->ato.state == UI_ATO_STATE_OVERFLOW ||
                  vm->ato.state == UI_ATO_STATE_BLOCKED ||
                  vm->ato.state == UI_ATO_STATE_ERROR ||
                  vm->ato.state == UI_ATO_STATE_DISABLED;

    if (unsafe) lv_obj_add_state(btn_refill, LV_STATE_DISABLED);
    else lv_obj_clear_state(btn_refill, LV_STATE_DISABLED);
}

6. Criar teste textual de baseline
Criar script simples `tools/check_normative_strings.py`:

```python
from pathlib import Path
import sys

forbidden = ["PZEM-004T v3.0"]
roots = [Path("firmware/main"), Path("firmware/docs"), Path("firmware/ARCHITECTURE.md")]
failed = False

for root in roots:
    files = [root] if root.is_file() else list(root.rglob("*"))
    for f in files:
        if not f.is_file():
            continue
        if f.suffix.lower() not in [".c", ".h", ".md", ".txt"]:
            continue
        text = f.read_text(errors="ignore")
        for token in forbidden:
            if token in text:
                print(f"FORBIDDEN TOKEN {token} in {f}")
                failed = True

sys.exit(1 if failed else 0)
```

4. Sugestão de PowerShell para escopo auditável

```powershell
param(
    [string]$Root = ".",
    [string]$Out = "auditoria_codigo_proprio.txt"
)

$includeExtensions = @(".c", ".h", ".cpp", ".hpp", ".md", ".txt", ".cmake")
$excludeDirs = @(
    "managed_components",
    "build",
    ".git",
    ".vscode",
    "node_modules"
)
$excludeExtensions = @(
    ".png", ".jpg", ".jpeg", ".gif", ".bmp", ".ttf", ".bin", ".elf", ".map", ".o", ".a", ".zip"
)

if (Test-Path $Out) { Remove-Item $Out -Force }

$files = Get-ChildItem -Path $Root -Recurse -File | Where-Object {
    $full = $_.FullName
    $ext = $_.Extension.ToLowerInvariant()

    foreach ($dir in $excludeDirs) {
        if ($full -match "[\\/]$([regex]::Escape($dir))[\\/]") { return $false }
    }

    if ($excludeExtensions -contains $ext) { return $false }
    return $includeExtensions -contains $ext
} | Sort-Object FullName

"TOTAL_ARQUIVOS_AUDITAVEIS=$($files.Count)" | Out-File -FilePath $Out -Encoding UTF8

foreach ($f in $files) {
    "`n===== FILE: $($f.FullName) =====`n" | Out-File -FilePath $Out -Append -Encoding UTF8
    Get-Content $f.FullName -Raw -ErrorAction SilentlyContinue | Out-File -FilePath $Out -Append -Encoding UTF8
}

Write-Host "Arquivo gerado: $Out"
Write-Host "Total de arquivos auditáveis: $($files.Count)"
```
